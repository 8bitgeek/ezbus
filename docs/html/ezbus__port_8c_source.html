80

89 static int ezbus_private_recv( ezbus_port_t* port, void* buf, uint32_t index, size_t size )
90 {
91  int ch;
92  uint8_t* p = (uint8_t*)buf;
93  ezbus_ms_tick_t start = ezbus_platform_get_ms_ticks();
94  /* receive the entire header or timeout... */
95  while ( index < size && (ezbus_platform_get_ms_ticks() - start) <= port->packet_timeout )
96  {
97  if ( (ch = ezbus_port_getch(port)) >= 0 )
98  {
99  p[index++] = ch;
100  start = ezbus_platform_get_ms_ticks();
101  }
102  }
103  return index;
104 }

116 extern EZBUS_ERR ezbus_port_recv( ezbus_port_t* port, ezbus_packet_t* packet )
117 {
118  EZBUS_ERR err = EZBUS_ERR_NOTREADY;
119  int index = 0;
120  uint8_t* p = (uint8_t*)&packet->header;
121  int ch;
122 
123  ch = ezbus_seek_leadin( port );
124  if ( ch == EZBUS_MARK )
125  {
126  p[ index++ ] = ch;
127  index = ezbus_private_recv( port, p, index, sizeof( ezbus_header_t ) );
128  if ( index == sizeof( ezbus_header_t ) )
129  {
130  ezbus_packet_header_flip( packet );
131  if ( ezbus_packet_header_valid_crc( packet ) )
132  {
133  if ( ezbus_packet_data_size( packet ) )
134  {
135  index = ezbus_private_recv( port, &packet->data.crc, 0, sizeof( ezbus_crc_t ) );
136  if ( index == sizeof( ezbus_crc_t ) )
137  {
138  index = ezbus_private_recv( port, ezbus_packet_data( packet ), 0, ezbus_packet_data_size( packet ) );
139  if ( index == ezbus_packet_data_size( packet ) )
140  {
141  ezbus_packet_data_flip( packet );
142  err = ezbus_packet_data_valid_crc( packet ) ? EZBUS_ERR_OKAY : EZBUS_ERR_CRC;
143  }
144  else
145  {
146  err = EZBUS_ERR_TIMEOUT;
147  }
148  }
149  else
150  {
151  err = EZBUS_ERR_CRC;
152  }
153  }
154  else
155  {
156  err = EZBUS_ERR_OKAY;
157  }
158  }
159  else
160  {
161  err = EZBUS_ERR_CRC;
162  }
163  }
164  else
165  {
166  err = EZBUS_ERR_TIMEOUT;
167  }
168  }
169 
170  if ( err == EZBUS_ERR_OKAY )
171  {
172  ezbus_hex_dump( "RX:", packet, ezbuf_packet_bytes_to_send( packet ) );
173  }
174 
175  return err;
176 }

205 uint32_t ezbus_port_byte_time_ns( ezbus_port_t* port )
206 {
207  uint32_t bits_sec = port->speed;
208  uint32_t nsec_bit = 1000000000 / bits_sec; /* ex. 1000000000 / 10000000 = 100 */
209  uint32_t nsec_byte = nsec_bit * 10; /* ex. 100 * 10 = 1000 */
210  return nsec_byte;
211 }

213 uint32_t ezbus_port_packet_timeout_time_ms( ezbus_port_t* port )
214 {
215  /* 1000000 ns in a ms. */
216  uint32_t nsec_byte = ezbus_port_byte_time_ns(port);
217  uint32_t nsec_packet = sizeof(ezbus_packet_t) * nsec_byte;
218  uint32_t msec_packet = nsec_packet/1000000;
219  return msec_packet?msec_packet:1;
220 }
