38 extern bool ezbus_socket_callback_transmitter_empty( ezbus_mac_t* mac )
39 {
40  /*
41  * The mac transmitter buffer has become available.
42  * attempt to give all ports a fair shake at transmitting
43  */
44  for( int n=0; n < ezbus_socket_max(); n++ )
45  {
46  ezbus_socket_t socket = ezbus_socket_cycle_next();
47  if ( ezbus_socket_is_open( socket ) )
48  {
49  if ( ezbus_socket_callback_send ( next_tx_socket ) )
50  {
51  return true;
52  }
53  }
54  }
55  return false;
56 }

58 static ezbus_socket_t ezbus_socket_cycle_next( void )
59 {
60  if ( ++next_tx_socket >= ezbus_socket_max() )
61  next_tx_socket = 0;
62  return next_tx_socket;
63 }

86 {
87  ezbus_packet_t* rx_packet = ezbus_mac_get_receiver_packet( mac );
88  ezbus_address_t* peer = ezbus_packet_src( rx_packet );
89  ezbus_socket_t socket = ezbus_packet_dst_socket( rx_packet );
90 
91  if ( socket == EZBUS_SOCKET_ANY )
92  {
93  socket = ezbus_socket_open( mac, peer );
94  }
95
