48 int ezbus_platform_open(ezbus_platform_port_t* port,uint32_t speed)
49 {
50  if ( (port->fd = open(port->serial_port_name,O_RDWR)) >=0 )
51  {
52  struct serial_rs485 rs485conf = {0};
53 
54  rs485conf.flags |= SER_RS485_ENABLED;
55  rs485conf.flags |= SER_RS485_RTS_ON_SEND;
56  rs485conf.flags &= ~(SER_RS485_RTS_AFTER_SEND);
57 
58  ioctl (port->fd, TIOCSRS485, &rs485conf);
59  ezbus_platform_set_speed(port,speed);
60  serial_set_blocking(port->fd,false);
61  return 0;
62  }
63  perror("ezbus_platform_open");
64  return -1;
65 }

107 int ezbus_platform_set_speed(ezbus_platform_port_t* port,uint32_t baud)
108 {
109  struct termios2 options;
110 
111  if ( port->fd >= 0 )
112  {
113  fcntl(port->fd, F_SETFL, 0);
114 
115  ioctl(port->fd, TCGETS2, &options);
116 
117  options.c_cflag &= ~CBAUD;
118  options.c_cflag |= BOTHER;
119  options.c_ispeed = baud;
120  options.c_ospeed = baud;
121 
122  options.c_cflag |= (CLOCAL | CREAD);
123  options.c_cflag &= ~CRTSCTS;
124  options.c_cflag &= ~HUPCL;
125 
126  options.c_iflag &= ~(IGNBRK | BRKINT | PARMRK | ISTRIP | INLCR | IGNCR | ICRNL | IXON);
127  options.c_oflag &= ~OPOST;
128  options.c_lflag &= ~(ECHO | ECHONL | ICANON | ISIG | IEXTEN);
129  options.c_cflag &= ~(CSIZE | PARENB);
130  options.c_cflag |= CS8;
131 
132  ioctl(port->fd, TCSETS2, &options);
133 
134  return 0;
135  }
136  return -1;
137 }

228 static void serial_set_blocking (int fd, int should_block)
229 {
230  struct termios2 options;
231 
232  if ( fd >= 0 )
233  {
234  fcntl(fd, F_SETFL, 0);
235 
236  ioctl(fd, TCGETS2, &options);
237 
238  if ( should_block )
239  {
240  options.c_cc[VMIN] = 1;
241  options.c_cc[VTIME] = 1;
242  }
243  else
244  {
245  options.c_cc[VMIN] = 0;
246  options.c_cc[VTIME] = 0;
247  }
248 
249  ioctl(fd, TCSETS2, &options);
250  }
251 }
