222 extern uint16_t ezbus_packet_data_size( ezbus_packet_t* packet )
223 {
224  uint16_t size=0;
225  switch ( ezbus_packet_type( packet ) )
226  {
227  case packet_type_reset:
228  case packet_type_take_token:
229  size = sizeof( ezbus_token_t );
230  break;
231  case packet_type_give_token:
232  size = sizeof( ezbus_token_t );
233  break;
234  case packet_type_ack:
235  case packet_type_nack:
236  case packet_type_coldboot:
237  case packet_type_warmboot_rq:
238  case packet_type_warmboot_rp:
239  case packet_type_warmboot_ak:
240  break;
241  case packet_type_parcel:
242  size = sizeof( ezbus_parcel_t );
243  break;
244  case packet_type_speed:
245  size = sizeof( ezbus_speed_t );
246  break;
247  }
248  return size;
249 }
