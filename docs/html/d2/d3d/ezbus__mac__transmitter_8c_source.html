38 void ezbus_mac_transmitter_run ( ezbus_mac_t* mac )
39 {
40  static ezbus_mac_transmitter_state_t transmitter_state=(ezbus_mac_transmitter_state_t)0xff;
41 
42  if ( ezbus_mac_transmitter_get_state( mac ) != transmitter_state )
43  {
44  ezbus_log( EZBUS_LOG_TRANSMITTERSTATE, "%s\n", ezbus_mac_transmitter_get_state_str(mac) );
45  transmitter_state = ezbus_mac_transmitter_get_state( mac );
46  }
47 
48  switch( ezbus_mac_transmitter_get_state( mac ) )
49  {
50 
51  case transmitter_state_empty:
52  ezbus_mac_transmitter_signal_empty( mac );
53  break;
54 
55  case transmitter_state_full:
56  ezbus_mac_transmitter_signal_full( mac );
57  ezbus_mac_transmitter_set_state( mac, transmitter_state_send );
58  break;
59 
60  case transmitter_state_send:
61  do_mac_transmitter_state_send( mac );
62  break;
63 
64  case transmitter_state_sent:
65  do_mac_transmitter_state_sent( mac );
66  break;
67  }
68 }

100 static void do_mac_transmitter_state_send( ezbus_mac_t* mac )
101 {
102  ezbus_mac_transmitter_set_err( mac, ezbus_port_send( ezbus_mac_get_port( mac ), ezbus_mac_get_transmitter_packet( mac ) ) );
103  if ( ezbus_mac_transmitter_get_err( mac ) == EZBUS_ERR_OKAY )
104  {
105  ezbus_mac_transmitter_set_state( mac, transmitter_state_sent );
106  }
107  else
108  {
109  ezbus_mac_transmitter_signal_fault( mac );
110  }
111 }

113 static void do_mac_transmitter_state_sent( ezbus_mac_t* mac )
114 {
115  ezbus_packet_t* tx_packet = ezbus_mac_get_transmitter_packet( mac );
116 
117  ezbus_mac_transmitter_signal_sent( mac );
118  if ( ezbus_packet_type( tx_packet ) == packet_type_parcel )
119  {
120  if ( ezbus_packet_ack_req( tx_packet ) )
121  {
122  ezbus_mac_transmitter_signal_wait( mac );
123  }
124  }
125  ezbus_mac_transmitter_reset( mac );
126 }
