47 extern ezbus_socket_t ezbus_socket_open( ezbus_mac_t* mac, ezbus_address_t* peer )
48 {
49  ezbus_socket_t rc = ezbus_socket_slot_available();
50  if ( rc != EZBUS_SOCKET_INVALID )
51  {
52  ezbus_socket_state_t* socket_state = &ezbus_sockets[rc];
53  ezbus_platform_memset( socket_state, 0, sizeof(ezbus_socket_state_t) );
54  socket_state->mac = mac;
55  socket_state->peer = peer;
56  ++socket_count;
57  }
58  return rc;
59 }

124 static ezbus_socket_t ezbus_socket_slot_available( void )
125 {
126  if ( ezbus_socket_count() < ezbus_socket_max() )
127  {
128  for( ezbus_socket_t n=0; n < ezbus_socket_max(); n++ )
129  {
130  if ( ezbus_socket_mac( n ) == NULL )
131  {
132  return n;
133  }
134  }
135  }
136  return EZBUS_SOCKET_INVALID;
137 }
