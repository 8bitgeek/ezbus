37 extern void ezbus_mac_warmboot_init( ezbus_mac_t* mac )
38 {
39  ezbus_mac_warmboot_t* boot = ezbus_mac_get_warmboot( mac );
40 
41  ezbus_platform_memset( boot, 0 , sizeof( ezbus_mac_warmboot_t) );

52 extern void ezbus_mac_warmboot_run( ezbus_mac_t* mac )
53 {
54  ezbus_mac_warmboot_t* boot = ezbus_mac_get_warmboot( mac );
55  ezbus_timer_run( &boot->warmboot_timer );
56  switch ( ezbus_mac_warmboot_get_state( mac ) )
57  {
58  case state_warmboot_idle: do_state_warmboot_idle ( mac ); break;
59  case state_warmboot_start: do_state_warmboot_start ( mac ); break;
60  case state_warmboot_continue: do_state_warmboot_continue ( mac ); break;
61  case state_warmboot_stop: do_state_warmboot_stop ( mac ); break;
62  case state_warmboot_finished: do_state_warmboot_finished ( mac ); break;
63  }
64 }

142 static void ezbus_mac_warmboot_period_timeout( ezbus_timer_t* timer, void* arg )
143 {
144  ezbus_mac_t* mac = (ezbus_mac_t*)arg;
145  ezbus_mac_warmboot_t* boot = ezbus_mac_get_warmboot( mac );
146  ezbus_timer_stop( &boot->warmboot_timer );
147 
148  ezbus_mac_arbiter_dec_warmboot_cycles( mac );
149  if ( ezbus_mac_arbiter_get_warmboot_cycles( mac ) == 0 )
150  {
151  ezbus_mac_warmboot_set_state( mac, state_warmboot_finished );
152  }
153  else
