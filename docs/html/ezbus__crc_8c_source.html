50 extern ezbus_crc_t* ezbus_crc( ezbus_crc_t* crc, void* p,size_t size )
51 {
52  uint8_t* d=(uint8_t*)p;
53  for(size_t n=0; n < size; n++)
54  {
55  ezbus_crc_update(crc,*d++);
56  }
57  return crc;
58 }

60 extern ezbus_crc_t* ezbus_crc_update( ezbus_crc_t* crc, uint8_t c )
61 {
62  uint16_t tmp, short_c;
63  short_c = 0x00ff & (uint16_t) c;
64  tmp = (crc->word) ^ short_c;
65  crc->word = ((crc->word) >> 8) ^ crc_tab[ tmp & 0xff ];
66  return crc;
67 }
