90 {
91  ezbus_mac_receiver_signal_empty( mac );
92  ezbus_mac_receiver_set_err( mac, ezbus_port_recv( ezbus_mac_get_port( mac ), ezbus_mac_get_receiver_packet( mac ) ) );
93 
94  if ( ezbus_mac_receiver_get_err( mac ) == EZBUS_ERR_OKAY )
