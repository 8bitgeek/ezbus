172 static void do_state_coldboot_start( ezbus_mac_t* mac )
173 {
174  ezbus_mac_coldboot_t* boot = ezbus_mac_get_coldboot( mac );
175  ezbus_timer_stop( &boot->silent_timer );
176  ezbus_timer_stop( &boot->coldboot_timer );
177  ezbus_mac_coldboot_signal_start( mac );
178  ezbus_timer_set_period (
179  &boot->coldboot_timer,
180  ezbus_platform_random( EZBUS_COLDBOOT_TIMER_MIN, EZBUS_COLDBOOT_TIMER_MAX )
181  );
182 
183  //fprintf( stderr, "%d\n",ezbus_timer_get_period(&boot->coldboot_timer));
184 
185  ezbus_timer_start( &boot->coldboot_timer );
186  ezbus_mac_coldboot_set_state( mac, state_coldboot_continue );
187 }

189 static void do_state_coldboot_continue( ezbus_mac_t* mac )
190 {
191  ezbus_mac_coldboot_t* boot = ezbus_mac_get_coldboot( mac );
192  ++boot->seq;
193  /* If I'm the "last man standing" then seize control of the bus */
194  if ( ezbus_mac_coldboot_get_emit_count( boot ) > EZBUS_COLDBOOT_CYCLES )
195  {
196  ezbus_timer_stop( &boot->coldboot_timer );
197  ezbus_mac_coldboot_set_state( mac, state_coldboot_dominant );
198  }
199 }

216 static void ezbus_mac_coldboot_timer_callback( ezbus_timer_t* timer, void* arg )
217 {
218  ezbus_mac_t* mac = (ezbus_mac_t*)arg;
219  ezbus_mac_coldboot_t* boot = ezbus_mac_get_coldboot( mac );
220  if ( ezbus_timer_expired( timer ) )
221  {
222  ezbus_mac_coldboot_signal_continue( mac );
223  ezbus_mac_coldboot_inc_emit_count( boot );
224  ezbus_mac_coldboot_set_state( mac, state_coldboot_start );
225  }
226 }
