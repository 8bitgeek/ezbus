183 static void do_mac_arbiter_state_online( ezbus_mac_t* mac )
184 {
185  static bool online=false;
186 
187  if ( ezbus_mac_token_acquired( mac ) != online )
188  {
189  online = ezbus_mac_token_acquired( mac );
190  ezbus_log( EZBUS_LOG_ONLINE, "online %s\n", online ? "++TOKEN++" : "---------" );
191  }
192 
193  if ( ezbus_mac_token_acquired( mac ) )
194  {
195  if ( ezbus_mac_transmitter_empty( mac ) )
196  {
197  if ( !ezbus_mac_arbiter_transmit_busy( mac ) )
198  {
199  ezbus_log( EZBUS_LOG_ONLINE, "online tx parcel\n" );
200  ezbus_socket_callback_transmitter_empty( mac );
201  }
202  }
203 
204  if ( ezbus_mac_transmitter_empty( mac ) )
205  {
206  ezbus_mac_arbiter_t* arbiter = ezbus_mac_get_arbiter( mac );
207  if ( arbiter->rx_ack_pend )
208  {
209  ezbus_mac_arbiter_ack_parcel( mac, arbiter->rx_ack_seq, &arbiter->rx_ack_address );
210  arbiter->rx_ack_pend=false;
211  arbiter->rx_ack_seq=0;
212  ezbus_address_copy(&arbiter->rx_ack_address,&ezbus_broadcast_address);
213  }
214  else
215  {
216  ezbus_log( EZBUS_LOG_ONLINE, "online tx tok\n" );
217  ezbuz_mac_arbiter_transmit_token( mac );
218  ezbus_mac_token_relinquish( mac );
219  }
220  }
221  }

231  ezbus_mac_token_reset( mac );

242 static void ezbuz_mac_arbiter_receive_token( ezbus_mac_t* mac, ezbus_packet_t* packet )
243 {
244  ezbus_crc_t crc;
245 
246  ezbus_log( EZBUS_LOG_TOKEN, "ezbuz_mac_arbiter_receive_token\n" );
247 
248  ezbus_mac_peers_crc( mac, &crc );
249  ezbus_mac_arbiter_set_token_age( mac, ezbus_packet_get_token_age(packet) );
250  if ( ezbus_crc_equal( &crc, ezbus_packet_get_token_crc( packet ) ) )
251  {
252  ezbus_log( EZBUS_LOG_TOKEN, "ezbus_mac_token_acquire\n" );
253  ezbus_mac_token_acquire( mac );
254 
255  if ( ezbus_packet_get_token_age(packet) > EZBUS_WARMBOOT_AGE )
256  {
257  ezbus_log( EZBUS_LOG_TOKEN, "ezbus_packet_get_token_age\n" );
258  ezbus_mac_arbiter_set_token_age( mac, 0 );
259  ezbus_mac_arbiter_set_state( mac, mac_arbiter_state_reboot_warm );
260  do_mac_arbiter_state_reboot_warm( mac );
261  }
262  }
263  else
264  {
265  ezbus_mac_arbiter_set_state( mac, mac_arbiter_state_reboot_warm );
266  }
267 }

274 extern void ezbus_mac_arbiter_receive_signal_parcel( ezbus_mac_t* mac, ezbus_packet_t* packet )
275 {
276 
277  if ( ezbus_address_compare( ezbus_packet_dst(packet), &ezbus_self_address ) == 0 )
278  {
279  ezbus_mac_arbiter_t* arbiter = ezbus_mac_get_arbiter( mac );
280  ezbus_packet_t* rx_packet = ezbus_mac_get_receiver_packet( mac );
281 
282  ezbus_log( EZBUS_LOG_RECEIVER, "do_receiver_packet_type_parcel\n" );
283 
284  if ( !arbiter->rx_ack_pend )
285  {
286  ezbus_socket_callback_receiver_ready( mac, packet );
287 
288  arbiter->rx_ack_pend = true;
289  arbiter->rx_ack_seq = ezbus_packet_seq( rx_packet );
290  ezbus_address_copy( &arbiter->rx_ack_address, ezbus_packet_src( rx_packet ) );
291  }
292  }
293 }

295 static void ezbus_mac_arbiter_ack_parcel( ezbus_mac_t* mac, uint8_t seq, ezbus_address_t* address )
296 {
297  ezbus_packet_t tx_packet;
298  ezbus_packet_t* rx_packet = ezbus_mac_get_receiver_packet( mac );
299 
300  ezbus_packet_init ( &tx_packet );
301  ezbus_packet_set_type ( &tx_packet, packet_type_ack );
302  ezbus_packet_set_dst_socket ( &tx_packet, ezbus_packet_src_socket( rx_packet ) );
303  ezbus_packet_set_src_socket ( &tx_packet, ezbus_packet_dst_socket( rx_packet ) );
304  ezbus_packet_set_seq ( &tx_packet, ezbus_packet_seq( rx_packet ) );
305  ezbus_packet_set_src ( &tx_packet, &ezbus_self_address );
306  ezbus_packet_set_dst ( &tx_packet, ezbus_packet_src( rx_packet ) );
307 
308  ezbus_mac_transmitter_put( mac, &tx_packet );
309 }
