33 extern void ezbus_timer_run( ezbus_timer_t* timer )
34 {
35  switch( timer->state )
36  {
37  case state_timer_stopping:
38  ezbus_timer_set_state( timer, state_timer_stopped );
39  break;
40  case state_timer_stopped:
41  break;
42  case state_timer_starting:
43  timer->start = ezbus_timer_get_ticks( timer );
44  ezbus_timer_set_state( timer, state_timer_started );
45  break;
46  case state_timer_started:
47  if ( ezbus_timer_timeout( timer ) )
48  ezbus_timer_set_state( timer, state_timer_expiring );
49  break;
50  case state_timer_pausing:
51  ezbus_timer_set_state( timer, state_timer_paused );
52  break;
53  case state_timer_paused:
54  break;
55  case state_timer_resuming:
56  ezbus_timer_set_state( timer, state_timer_resume );
57  break;
58  case state_timer_resume:
59  ezbus_timer_set_state( timer, state_timer_started );
60  break;
61  case state_timer_expiring:
62  ezbus_timer_set_state( timer, state_timer_expired );
63  break;
64  case state_timer_expired:
65  ezbus_log( EZBUS_LOG_TIMERS, "state_timer_expired - %s\n", eabus_timer_get_key( timer ) );
66  timer->callback( timer, timer->arg );
67  break;
68  }
69 }
