65 extern EZBUS_ERR ezbus_mac_peers_insort( ezbus_mac_t* mac, const ezbus_peer_t* peer )
66 {
67  EZBUS_ERR err = EZBUS_ERR_OKAY;
68 
69  if ( !ezbus_mac_peers_full( mac) )
70  {
71  if ( ezbus_address_compare( ezbus_peer_get_address( peer ), (ezbus_address_t*)&ezbus_warmboot_address ) != 0 &&
72  ezbus_address_compare( ezbus_peer_get_address( peer ), (ezbus_address_t*)&ezbus_broadcast_address ) != 0)
73  {
74  if ( ezbus_mac_peers_index_of( mac, ezbus_peer_get_address( peer ) ) < 0 )
75  {
76  for( int index=0; index < ezbus_mac_peers_count( mac ); index++ )
77  {
78  ezbus_address_t* other_address = ezbus_peer_get_address( ezbus_mac_peers_at( mac, index ) );
79  ezbus_address_t* peer_address = ezbus_peer_get_address( peer );
80 
81  if ( ezbus_address_compare( peer_address, other_address ) < 0 )
82  {
83  err = ezbus_mac_peers_insert( mac, peer, index );
84  return err;
85  }
86  }
87  err = ezbus_mac_peers_append( mac, peer );
88  }
89  }
90  }
91  else
92  {
93  err = EZBUS_ERR_LIMIT;
94  }
95 
96  return err;
97 }

215 static EZBUS_ERR ezbus_mac_peers_insert( ezbus_mac_t* mac, const ezbus_peer_t* peer, int index )
216 {
217  EZBUS_ERR err = EZBUS_ERR_LIMIT;
218  if ( !ezbus_mac_peers_full( mac) )
219  {
220  if ( index >= ezbus_mac_peers_count( mac ) )
221  {
222  err = ezbus_mac_peers_append( mac, peer );
223  }
224  else
225  {
226  ezbus_mac_peers_t* peers = ezbus_mac_get_peers( mac );
227  int peers_to_move = ( ezbus_mac_peers_count( mac ) - index );
228  int bytes_to_move = ( peers_to_move * sizeof(ezbus_peer_t) );
229  ezbus_peer_t* move_dst = &peers->list[index+1];
230  ezbus_peer_t* dst = &peers->list[index];
231 
232  ezbus_platform_memmove( move_dst, dst, bytes_to_move );
233  ezbus_peer_copy( dst, peer );
234 
235  ++peers->count;
236  }
237  err = EZBUS_ERR_OKAY;
238  }
239  return err;
240 }
