180 static void ezbus_arbiter_ack_tx_timer_triggered( ezbus_timer_t* timer, void* arg )
181 {
182  ezbus_mac_t* mac = (ezbus_mac_t*)arg;
183  ezbus_mac_arbiter_transmit_t* arbiter_transmit = ezbus_mac_get_arbiter_transmit( mac );
184 
185  ezbus_log( EZBUS_LOG_ARBITER, "ezbus_arbiter_ack_tx_timer_triggered\n" );
186 
187  if ( arbiter_transmit->ack_tx_count-- > 0 )
188  {
189  if ( ezbus_socket_callback_transmitter_resend( mac ) )
190  {
191  ezbus_timer_restart( &arbiter_transmit->ack_tx_timer );
192  }
193  else
194  {
195  ezbus_mac_arbiter_transmit_reset( mac );
196  }
197  }
198  else
199  {
200  ezbus_mac_arbiter_transmit_reset( mac );
201  ezbus_socket_callback_transmitter_limit( mac );
202  }
203 }
