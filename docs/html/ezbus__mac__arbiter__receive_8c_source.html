112 static void do_receiver_packet_type_ack( ezbus_mac_t* mac, ezbus_packet_t* packet )
113 {
114  if ( ezbus_address_compare( ezbus_packet_dst(packet), &ezbus_self_address ) == 0 )
115  {
116  ezbus_packet_t* tx_packet = ezbus_mac_get_transmitter_packet( mac );
117  if ( ezbus_address_compare( ezbus_packet_src(packet), ezbus_packet_dst(tx_packet) ) == 0 )
118  {
119  if ( ezbus_packet_seq(packet) == ezbus_packet_seq(tx_packet) )
120  {
121  ezbus_mac_arbiter_transmit_reset( mac );
122  ezbus_socket_callback_transmitter_ack( mac );
123  }
124  else
125  {
126  ezbus_socket_callback_transmitter_fault( mac );
127  ezbus_log( EZBUS_LOG_ARBITER, "recv: ack seq mismatch\n");
128  }
129  }
130  else
131  {
132  ezbus_socket_callback_transmitter_fault( mac );
133  ezbus_log( EZBUS_LOG_ARBITER, "recv: ack address mismatch\n" );
134  }
135  }

165 static void do_receiver_packet_type_coldboot( ezbus_mac_t* mac, ezbus_packet_t* packet )
166 {
167  ezbus_peer_t peer;
168  ezbus_mac_coldboot_t* boot = ezbus_mac_get_coldboot( mac );
169  ezbus_mac_arbiter_receive_t* arbiter_receive = ezbus_mac_get_arbiter_receive( mac );
170 
171  ezbus_mac_warmboot_set_state( mac, state_warmboot_idle );
172 
173  if ( (int)ezbus_mac_coldboot_get_state( mac ) < (int)state_coldboot_silent_stop )
174  {
175  ezbus_mac_coldboot_set_state( mac, state_coldboot_silent_stop );
176  }
177 
178  ezbus_log( EZBUS_LOG_BOOTSTATE, "%ccoldboot <%s %3d | ", ezbus_mac_token_acquired(mac)?'*':' ', ezbus_address_string( ezbus_packet_src( packet ) ), ezbus_packet_seq( packet ) );
179  ezbus_mac_peers_log( mac );
180 
181  arbiter_receive->warmboot_seq=0;
182  ezbus_peer_init( &peer, ezbus_packet_src( packet ), ezbus_packet_seq( packet ) );
183 
184  if ( ezbus_address_compare( &ezbus_self_address, ezbus_packet_src( packet ) ) > 0 )
185  {
186  ezbus_timer_stop( &boot->coldboot_timer );
187  ezbus_mac_coldboot_set_state( mac, state_coldboot_silent_start );
188  }

200

202  {

204  {

211  }

214  ezbus_timer_stop( &arbiter_receive->warmboot_timer );

295 extern void ezbus_mac_receiver_signal_full ( ezbus_mac_t* mac )
296 {

308  case packet_type_ack: do_receiver_packet_type_ack ( mac, packet ); break;
309  case packet_type_nack: do_receiver_packet_type_nack ( mac, packet ); break;
310  case packet_type_coldboot: do_receiver_packet_type_coldboot ( mac, packet ); break;
311  case packet_type_warmboot_rq: do_receiver_packet_type_warmboot_rq ( mac, packet ); break;
312  case packet_type_warmboot_rp: do_receiver_packet_type_warmboot_rp ( mac, packet ); break;
313  case packet_type_warmboot_ak: do_receiver_packet_type_warmboot_ak ( mac, packet ); break;
314  }
315 
316  ezbus_mac_arbiter_receive_sniff( mac, packet );

318  ezbus_mac_peers_log( mac );
