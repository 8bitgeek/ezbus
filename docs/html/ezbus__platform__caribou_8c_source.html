42 int ezbus_platform_send(ezbus_platform_port_t* port,void* bytes,size_t size)
43 {
44  uint8_t* p = (uint8_t*)bytes;
45  size_t sent=0;
46  do {
47  sent += fwrite(p,1,size-sent,port->fd);
48  p = (uint8_t*)bytes;
49  p = &p[sent];
