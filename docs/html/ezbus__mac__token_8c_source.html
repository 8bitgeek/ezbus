49 extern uint32_t ezbus_mac_token_ring_time( ezbus_mac_t* mac )
50 {
51  uint32_t packet_sz = sizeof(ezbus_header_t);
52  #if NUM_PEERS_HACK
53  uint32_t packets_per_round = EZBUS_MAX_PEERS * 2;
54  #else
55  uint32_t packets_per_round = ( ezbus_mac_peers_count(mac) * 2);
56  #endif
57  float bit_time_sec = 1.0f/(float)ezbus_port_get_speed( ezbus_mac_get_port(mac) );
58  float packet_bits = ((float)packet_sz * 12.0f);
59  float packet_time_sec = packet_bits * bit_time_sec;
60  float secs_per_round = packet_time_sec * (float)packets_per_round;
61  return (int)((secs_per_round * 1000.0f) + 1.0f);
62 }
